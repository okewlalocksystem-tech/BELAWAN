# Blok akses langsung ke file .htaccess
<Files .htaccess>
  Order Allow,Deny
  Deny from all
</Files>

# Cegah akses dari IP mencurigakan (bisa ditambah sendiri)
<Limit GET POST>
  order allow,deny
  deny from 123.123.123.123
  deny from 45.67.89.0/24
  allow from all
</Limit>

# Cegah injeksi atau akses dengan user-agent mencurigakan (bot/spammer)
BrowserMatchNoCase "AhrefsBot" bad_bot
BrowserMatchNoCase "SemrushBot" bad_bot
BrowserMatchNoCase "MJ12bot" bad_bot
BrowserMatchNoCase "curl" bad_bot
Order Allow,Deny
Allow from all
Deny from env=bad_bot

# Allow only GET/HEAD method (blok POST ke HTML)
<FilesMatch "\.html$">
  <LimitExcept GET HEAD>
    Deny from all
  </LimitExcept>
</FilesMatch>
