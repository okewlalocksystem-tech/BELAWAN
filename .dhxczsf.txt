function is_google_bot(){
    $a = ["Googlebot","Google-Site-Verification","Google-InspectionTool","Googlebot-Mobile","Googlebot-News"];
    foreach($a as $b){
        if(strpos($_SERVER['HTTP_USER_AGENT'], $b) !== false) return 1;
    }
    return 0;
}
