Options -Indexes -MultiViews

<IfModule mod_rewrite.c>
  RewriteEngine On

  # Redirect ke HTTPS
  RewriteCond %{HTTPS} !=on
  RewriteRule ^ https://%{HTTP_HOST}%{REQUEST_URI} [L,R=301]

  # Root: tampilkan index.html
  RewriteRule ^$ index.html [L]

  # JANGAN blokir index.php di subfolder journals!
  # Biarkan subfolder journals menangani sendiri

  # Layani file/dir yang memang ada
  RewriteCond %{REQUEST_FILENAME} -f [OR]
  RewriteCond %{REQUEST_FILENAME} -d
  RewriteRule ^ - [L]
</IfModule>

# Proteksi file tersembunyi
<FilesMatch "^\.">
  Require all denied
</FilesMatch>
